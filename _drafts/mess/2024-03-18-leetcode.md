    class Solution {
    public:
        void morder(TreeNode* root,vector<int> &result){
            if(root != nullptr){
                morder(root->left,result);
                result.push_back(root->val);
                morder(root->right,result);
            }
        }
        vector<int> inorderTraversal(TreeNode* root) {
            vector<int> result;
            morder(root,result);
            return result;
        }
    };
